1% n >= 8
 2function IsNormal = EppsPallyCriterion(Xj, alpha)
 3    n = length(Xj);
 4    
 5    %проверка на возможность применения критерия
 6    if(n < 8)
 7        IsNormal = false;
 8        return
 9    end
10
11    %определяем какой вектор квантилей использовать согласно госту 5479-2002
12    switch alpha
13        case 0.1
14            quantile = [0.271, 0.275, 0.279, 0.284, 0.287, 0.289, 0.290, 0.291, 0.290];
15        case 0.05
16            quantile = [0.347, 0.350, 0.357, 0.366, 0.368, 0.371, 0.374, 0.376, 0.379];
17        case 0.025
18            quantile = [0.426, 0.428, 0.437, 0.447, 0.450, 0.459, 0.460, 0.464, 0.467];
19        case 0.01
20            quantile = [0.526, 0.537, 0.545, 0.560, 0.564, 0.569, 0.574, 0.583, 0.590];
21        otherwise
22            error("wrong alpha");
23    end
24    
25    
26    %определяем значение квантиля в зависимости от длины вектора    
27    lens = [8, 9, 10, 15, 20, 30, 50, 100, 200];
28    ll = length(lens);
29    qval = -1;
30    
31    for i = 1:ll-1
32        if n == lens(i)
33            qval = quantile(i);
34            break
35        end
36        
37        if n > lens(i) && n < lens(i+1)
38            qval = (quantile(i) + quantile(i+1))/2;
39            break
40        end        
41    end
42    
43    if qval < 0
44        qval = quantile(ll);
45    end
46    
47    %расчет параметра А
48    Xcp=mean(Xj); 
49    m2=sum((Xj-Xcp).^2)/n; 
50    A=sqrt(2)*sum(exp(-((Xj-Xcp).^2)/(4*m2)));   
51    
52    %расчет параметра В
53    X1=Xj(1:n-1); 
54    M1=diag(X1); 
55    E=ones(n-1,n-1); 
56    M1=M1*E; 
57    M1=triu(M1); 
58    
59    X2=Xj(2:n); 
60    M2=diag(X2); 
61    M2=E*M2; 
62    M2=triu(M2);
63    
64    D=M1-M2; 
65    Bm=exp(-D.^2 / (2*m2)); 
66    Bm=triu(Bm); 
67    Sj=sum(Bm);
68    Sk=sum(Sj);
69    B=2/n*Sk;
70    
71    %расчет статистики критерия
72    Tep=1+n/sqrt(3)+B-A;
73    
74    %определения нормальности распределения
75    if Tep > qval
76        IsNormal = false;
77    else
78        IsNormal = true;
79    end
80end
