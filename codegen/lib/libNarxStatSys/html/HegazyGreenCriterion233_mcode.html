1function IsNormal = HegazyGreenCriterion(Xj, alpha)
 2    %проверка на наличие известной апроксимации критических величин
 3    if ~(alpha == 0.01 || alpha == 0.05)
 4        error("invalid parametr alpha")
 5    end
 6    
 7    %расчет параметров для вычисления статистик T1 и T2 
 8    Xcp = mean(Xj);
 9    n = length(Xj);
10    
11    s = (Xj - Xcp);
12    s =  sqrt(sum(s .* s) / (n - 1));
13    
14    z = (Xj - Xcp) / s;
15    
16    %вектор мат ожидания порядковой статистики 
17    nu = zeros(1, n);
18    %надо заменить на более изящную конструкцию
19    for i = 1 : n
20        buf = n + 1;
21        nu(i) = 4.91 * ((i/buf)^0.14 - ((buf - i)/(buf))^0.14);
22    end
23    
24    %расчет статистик 
25    buf = z - nu;
26    t1 = sum(abs(buf)) / n;
27    t2 = sum(buf .* buf) / n;
28
29    %расчет критических значений в соответствие с известными апроксимациями
30    t1_crit = 0;
31    t2_crit = 0;
32    
33    if alpha == 0.01
34        t1_crit = 0.7195 - 0.1751 * log(n) + 0.0108 * log(n)^2;
35        t2_crit = 0.0178 + 2.8736/n - 8.2894/n^2;
36    end
37    
38    if alpha == 0.05
39        t1_crit = 0.6027 - 0.1481 * log(n) + 0.0090 * log(n)^2;
40        t2_crit = 0.0126 + 1.9227/n - 5.00677/n^2;
41    end
42    
43    %определение нормальности выборки
44    if t1 < t1_crit && t2 < t2_crit
45        IsNormal = true;
46    else
47        IsNormal = false;
48    end
49end
