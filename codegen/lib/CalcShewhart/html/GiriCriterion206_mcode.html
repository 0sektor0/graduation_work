1function IsNormal = GiriCriterion(Xj, alpha)
 2    %проверка на наличие известной апроксимации критических величин
 3    if ~(alpha == 0.05)
 4        error("invalid parametr alpha")
 5    end
 6    
 7    %расчет параметров, необходимых для дальнейших вычмслений
 8    Xcp = mean(Xj);
 9    n = length(Xj);
10    dsa = Xj - Xcp; %the Difference of the Sample and its Average. DSA 
11      
12    %расчет статистики
13    s =  sqrt(sum(dsa .* dsa)) / (n - 1);    
14    d = sum(abs(dsa)) / (n * s);
15    
16    %расчет мат ожид и дисперсии в соответсвие с их апроксимациями
17    exp = 0.797885 + 0.199471/n + 0.024934/n^2 - 0.031168/n^3 - 0.008182/n^4;
18    disp = 0.045070/n - 0.124648/n^2 + 0.084859/n^3 + 0.006323/n^4;
19    
20    %определене квантиля норм распределения заданного уровня
21    u = 0;
22    if alpha == 0.05
23        u = 1.645; %альфа квантиль стд нормального распределения уровня значимости 0.05 
24    end
25    
26    %определение нормальности распределения
27    d1 = exp + sqrt(disp)*(-u);
28    d2 = exp + sqrt(disp)*u;
29    
30    if d>d1 && d<d2
31        IsNormal = true;
32    else
33        IsNormal = false;
34    end
35end
