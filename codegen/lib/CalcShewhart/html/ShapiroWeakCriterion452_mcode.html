 2function IsNormal = ShapiroWeakCriterion(Xj) %для alpha = 0.05
 3    %расчет параметров для вычисления статистики W
 4    n = length(Xj); 
 5    m = n/2; 
 6    Xcp = mean(Xj);
 7        
 8    x = fliplr(Xj); 
 9    k = 1:m; 
10    zk = (n-2.*k+1) / (n-0.5); 
11    
12    a0 = 0.899 / ((n-2.4)^0.4162 - 0.02); 
13    ak = a0.*(zk + 1483./(3-zk).^10.845 + 71.6*10^-10./(1.1-zk).^8.26); 
14    
15    %расчет статистики
16    b=sum(ak.*(x(1:m)-Xj(1:m))).^2;
17    s2 = sum((Xj-Xcp).^2); 
18    w = (1-0.6695 / n^0.6518) * s2/b;
19
20    if(w < 1)
21        IsNormal = true;
22    else
23        IsNormal = false;
24    end
25end
